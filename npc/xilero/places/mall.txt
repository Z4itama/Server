function	script	funcdon8_1	{
set @menustr$,"";
mes getarg(0);
mes "If you have donated to XileRO already I will give you ^FF9900"+getarg(1)+"^000000 in exchange for your uber points!";
next;
set .max,getarraysize(getd("@"+getarg(2)));
if(.max > 20){
set .start,rand(.max-20);
set .end,.start+20;
}else{
set .start,0;
set .end,.max;
}

	for(set @i,.start;@i < .end;set @i,@i+1){
		if(compare(getd("@"+getarg(2)+"$["+@i+"]"),"+10")==1 || compare(getd("@"+getarg(2)+"$["+@i+"]"),"IFC")==1){
			set @menustr$,@menustr$+"^FF0000";
		}else if(compare(getarg(0),"Pets")==1){
			set @menustr$,@menustr$+"^0000FF";
		}else if(compare(getarg(0),"Epic")==1 || compare(getarg(0),"Mess")==1 || compare(getarg(0),"Clemente")==1 || compare(getarg(0),"Card Mas")==1){
			set @menustr$,@menustr$+"^008800";
		}else{
			set @menustr$,@menustr$+"^005555";
		}
		set @menustr$,@menustr$ + (getd("@" + getarg(2) + "$["+@i+"]"));
		if(@i < .end - 1){
				set @menustr$,@menustr$ + "^000000:";
		}
	}
	set @menustr$,@menustr$ + "^000000";
	set @c,select(@menustr$);
	set @c,(@c-1)+.start;
		mes getarg(0);
		mes (getitemname(getd("@"+getarg(2)+"["+@c+"]")))+" costs ^0000FF" + (getd("@"+getarg(2)+"p["+@c+"]"))+"^000000 Uber points";
		query_sql("SELECT `uberpoints` FROM `donationlog` WHERE `account_id`='"+getcharid(3)+"'",@uberpoints);
		dispbottom "Your total Uber Points:"+ @uberpoints + " UB";
		next;
		if(select("Wait, I'll think about it..","I'll take that one!")==2){
			if(@uberpoints >= getd("@"+getarg(2)+"p["+@c+"]")){
				mes getarg(0);
				mes "Thank you for donating, please come again!";

				query_sql("UPDATE `donationlog` SET `uberpoints`=`uberpoints`-"+(getd("@"+getarg(2)+"p["+@c+"]"))+" WHERE `account_id`='"+getcharid(3)+"'");
				if (compare(getd("@"+getarg(2)+"$["+@c+"]"),"+10")==1){
					getitem2(getd("@"+getarg(2)+"["+@c+"]"),1,1,10,0,0,0,0,0);
				}else{
					getitem(getd("@"+getarg(2)+"["+@c+"]"),1);
				}
				close;
			}
			else{
				mes getarg(0);
				mes "Sorry,but you need^0000FF "+ (getd("@"+getarg(2)+"p["+@c+"]")-@uberpoints) + " ^000000more Uber points to get this item!";
			}
		}else{
			mes @np$;
			mes "Okay, please take your time! For more information, ask the ^0000FFDonation Guide^000000 NPC. Thanks!";
		}
return;
}
